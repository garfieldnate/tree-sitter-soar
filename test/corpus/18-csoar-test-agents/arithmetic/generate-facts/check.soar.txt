   (state <s> ^operator <op>
              ^smem <smem>)
   (<op> ^name check)
   (<smem> ^result.failure <q>
           ^command <cmd>)
   (<cmd> ^query <q>)
-->
   (<s> ^phase check -
        ^phase store)
   (<cmd> ^query <q> -
          ^store <arithmetic>)

              ^smem.result.success <q>
              ^superstate <ss>)
   (<op> ^name check)
   (<ss> ^operator <ss-op>)
-->
   (<ss> ^arithmetic.facts existing)
}
