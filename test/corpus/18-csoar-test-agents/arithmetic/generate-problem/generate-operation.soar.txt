              ^name generate-problem
             -^problem)

   (<op> ^name generate-operation
         ^operation addition
         ^operation-symbol |+|)

              ^name generate-problem
             -^problem)

   (<op> ^name generate-operation
         ^operation subtraction
         ^operation-symbol |-|)
}

   (<c1> ^next-column <c2>
         ^column t)
   (<c2> ^next-column <c3>
         ^column t)
   (<c3> ^next-column nil
         ^column t)}
