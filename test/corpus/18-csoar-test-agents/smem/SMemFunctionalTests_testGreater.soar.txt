#Test the max query
sp {propose*query
  (state <s> ^superstate nil 
             ^smem <smem>)
  (<smem> -^result.retrieved)
--> 
  (<s> ^operator.name query)
}

sp {apply*query
  (state <s> ^operator.name query
             ^smem.command <sc>)
-->
  (<sc> ^query.name |val|)
  (<sc> ^math-query.value.greater 4)
}

sp {success
  (state <s> ^smem.result.retrieved <lti>)
  (<lti> ^name val
         ^value 5)
-->
  (succeeded)
}
