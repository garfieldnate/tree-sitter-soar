  (state <s> ^superstate nil 
             ^smem <smem>)

  (<s> ^operator.name query)
}

  (state <s> ^operator.name query
             ^smem.command <sc>)
-->
  (<sc> ^query.name |val|)
  (<sc> ^math-query.value.max <x>)
}

sp {success
  (state <s> ^smem.result.retrieved <lti>)
  (<lti> ^name val
         ^value 5
         ^value -2)
-->
  (succeeded)
}
