   (state <s> ^superstate nil

   (state <s> ^operator.name init-agent)

   (state <s> ^name smem-isupport-crash
              ^smem.command <command>)

   (state <s> ^name smem-isupport-crash
              ^smem.result <sres>)

   (<s> ^result <res>)
}

   (state <s> ^name smem-isupport-crash
              ^result.next <next>)
   (<next> -^value)

         ^lti <next>)
