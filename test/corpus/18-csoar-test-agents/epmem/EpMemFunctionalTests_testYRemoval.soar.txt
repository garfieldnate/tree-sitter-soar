# init-agent

sp {topstate*propose*init-agent
    (state <s> ^superstate nil
              -^name)
-->
    (<s> ^operator <op> +)
    (<op> ^name init-agent)
}

sp {topstate*apply*init-agent
    (state <s> ^operator.name init-agent)
-->
    (<s> ^topstate <s>
         ^name bolt-agent
         ^status add)
}

               ^status add)

}

               ^topstate.new-thing <seg>)

# add.create

    (state <s> ^name add
              -^interpreted)
-->
    (<s> ^operator.name create)
}

    (state <s> ^name add
               ^operator.name create
               ^topstate <ts>)
-->
    (<ts> ^new-thing.old1.old2 <help>)
    (<s> ^interpreted yes)
}

# add.store

    (state <s> ^name add
               ^new-thing <info>)
-->
    (<s> ^operator.name store)
}

sp {add*apply*store
    (state <s> ^name add
               ^operator.name store
               ^new-thing <info>
               ^topstate <ts>)
    (<ts> ^new-thing <new-thing>
          ^status add)
    (<new-thing> ^old1 <feature>)
-->
    (<ts> ^new-thing <info> -
          ^thing <thing>
          ^status add -
          ^status remove)
    (<thing> ^new1 <new-thing>
             ^new2 <feature>)
}
