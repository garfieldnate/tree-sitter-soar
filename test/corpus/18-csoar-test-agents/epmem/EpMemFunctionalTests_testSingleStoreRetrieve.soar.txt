   (state <s> ^name epmem-rand
              ^random)

   (state <s> ^operator <op>
              ^random <rand>
              ^epmem.command <cmd>)
   (<op> ^name remember)
   (<rand> ^num <num>)

   (write |Removing: | <num>)
   (<s> ^random <rand> -)
   (<cmd> ^query.random <remember-random>)
