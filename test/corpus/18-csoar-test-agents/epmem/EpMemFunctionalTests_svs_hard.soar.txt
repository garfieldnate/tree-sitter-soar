sp {propose*init
    (state <s> ^superstate nil
               -^name)
-->
    (<s> ^operator.name init)
}

sp {apply*init
    (state <s> ^operator.name init)
-->
    (<s> ^name svs
         ^state <state>)
    (<state> ^neg <neg1>
             ^neg <neg2>)
    (<x1> ^id x1)
    (<x2> ^id x2)
    (<neg1> ^var <x1>
            ^neg <x2>)
    (<neg2> ^var <x2>
            ^neg <x1>)
}

sp {svs*propose*retrieve
    (state <s> ^name svs)
-->
    (<s> ^operator.name retrieve)
}

sp {svs*apply*retrieve*query
    (state <s> ^operator.name retrieve
               ^epmem.command <cmd>)
-->
    (<cmd> ^query.state <q>)
    (<q> ^neg <neg1>
         ^neg <neg2>)
    (<x1> ^id x1)
    (<x2> ^id x2)
    (<neg1> ^var <x1>
            ^neg <x2>)
    (<neg2> ^var <x2>
            ^neg <x1>)
}

sp {svs*apply*retrieve*success
    (state <s> ^operator.name retrieve
               ^epmem <epmem>)
    (<epmem> ^command <cmd>
             ^result <res>)
    (<cmd> ^query <query>)
    (<res> ^success <query>
           ^match-cardinality 2)
-->
    (succeeded)
    (halt)
}

    (state <s> ^operator.name retrieve
               ^epmem <epmem>)
    (<epmem> ^command <cmd>
             ^result <res>)
    (<cmd> ^query <query>)
    (<res> ^success <query>
           ^match-cardinality {<> 2})
-->
    (halt)
}
