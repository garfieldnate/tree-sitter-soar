	:o-support
	(state <s> ^superstate nil
				-^collection)
-->
	(<s> ^collection <c>)
}

sp {propose*test
	(state <s> ^superstate nil)
-->
	(<s> ^operator <o> +)
	(<o> 	^name test
			^symbol <sym>)
}
sp {apply*test
	(state <s> ^operator.name test)
-->
	(<s> ^status done)
	(interrupt)
}

    (state <s> ^operator <o> +

-->

    (<u> ^operator <o>)					# OR RENAME THIS ATTRIBUTE
}										# (OR REPLACE ^used WITH (<c> ^operator <o>) )
