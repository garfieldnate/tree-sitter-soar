	:o-support
	(state <s> ^superstate nil
				-^collection)
-->
	(<s> ^collection <c>)
}

	(state <s> ^superstate nil)
-->
	(<s> ^operator <o> +)
	(<o> 	^name test
			^symbol <sym>)
}

	(state <s> ^operator.name test)
-->
	(<s> ^status done)
	(interrupt)
}
