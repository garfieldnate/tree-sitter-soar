  (state <s> ^superstate nil)
-->
  (<s> ^goal <goal>
       ^object <obj1> <obj2>)
  (<goal> ^finished false ^has-property <gp1> <gp2>)

  (<gp2> ^type room)

  (<obj2> ^type room)
}

  (state <s> ^superstate nil
             ^results <results>)
-->
  (<s> ^operator <o> + >)
  (<o> ^name finish-test)
}

sp {top-state*apply*finish-test
  (state <s> ^superstate nil
             ^goal <goal>
              ^operator.name finish-test)
    (<goal> ^finished false)
-->
    
    (<goal> ^finished false -)
    (<goal> ^finished true +)
    (interrupt)
}

  (state <s> ^superstate.superstate nil
              ^superstate <ss>)

        ^object <obj1>
        - ^results)
  (<goal> ^finished false
          ^has-property <p1>
          ^has-property {<p2> > <p1>})
  -->

       ^object <obj1>
       ^has-property <p1>
       ^has-property <p2>)
}

  (<o> ^name find-object-with-property
        ^has-property <desc>)
}

#  to an object in the world, 
# and create the grounded predicate on the results id

  (state <s> ^superstate.superstate nil
             ^operator <o>
             ^object <obj>
             ^substate-results <substate-results>)
  (<o> ^name find-object-with-property
       ^has-property <desc>)
  (<desc> ^<attr> <val1>)
  (<obj> ^<attr> <val1>)
-->
  (<s> ^found-property <desc>)
  (<substate-results> ^match <match-info>)
  (<match-info> ^has-property <desc> ^matched-object <obj> ^self-link-test <substate-results>)
}
