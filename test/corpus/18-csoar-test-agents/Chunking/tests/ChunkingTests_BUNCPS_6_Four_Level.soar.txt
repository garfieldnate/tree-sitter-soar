    (state <s3> ^superstate.superstate <ts>)

    (<o1> ^name level3 +
          ^aug <x>)

    (state <s1> ^superstate.superstate <ts>

          ^attr1 <y2>)
    -->
    (<ts> ^top-resulta <y>
          ^top-resultb <x>)
}

    (<o1> ^name <x>
          ^aug <y>)
-->
    (<ss> ^result3a <x>)
    (<ss> ^result3b <y>)
}
