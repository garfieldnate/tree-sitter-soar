   (state <s> ^object <block>
             -^ontop.bottom-block <block>)

   (<s> ^clear <block>)
}

# Table is always clear

sp {P2*elaborate*table*clear
   (state <s> ^object <object>)
   (<object> ^type table)
-->
   (<s> ^clear <object>)
}
