# while the other shouldn't be.

    (state <s> ^name substate
               ^operator.name test)
-->

    (state <s> ^name substate
               ^operator.name test
               ^superstate.test-item { > 8 < 10 >= <var-ungrounded> <var> }
               ^superstate.test-item2 { <= <var> <var2> }
               ^substate-output <var-ungrounded>
               ^superstate <ss>)
-->
    (<ss> ^result1a <var-ungrounded>
          ^result1b <var>
          ^result1c <var2>
          ^result1d <unbound>
          ^test-done 1)
    (<unbound> ^foo bar)
