   (state <s2> ^superstate.superstate.superstate nil)
-->
   (<s2> ^operator <o> + =)
   (<o> ^name make-chunk)}

   (state <s2> ^superstate <ts>)
   (<ts> ^superstate nil)
-->
   (force-learn <s2>)

}

   (state <s2> ^superstate.superstate <ts>)
   (<ts> ^superstate nil)
-->
   (<s2> ^top-state <ts>)
   (<s2> ^sub-struct <sub>)
   (<sub> ^foo1 bar1 ^foo2 bar2 ^sub-struct2 <sub2>)
   (<sub2> ^foo3 bar3)
}

    (state <s3> ^top-state <ts>)
    (<ts> ^value { > 1 <num> }
          ^foo <bar-copy>)
-->
    (<s3> ^foo <bar-copy>
          ^child <child>
          ^math (+ 20 <num>))
}

    (state <s> ^child <child>
               ^top-state.value <val>
               ^math <val2>
               ^sub-struct <sub>)
-->
    (<child> ^child-math (+ <val> <val2>)
             ^child-sub (deep-copy <sub>))
}

   (state <s4> ^operator <o2>
              ^math <number>
              ^child <child>
              ^top-state <ts>
              ^foo <bar>)
   (<ts> ^wme {<> <bar> <value> }
          ^value <new-num>)
   (<o2> ^name make-chunk)
-->
   (<ts> ^result <value> 
         ^new-math (* <number> <new-num>) 
         ^math (* <number> 2) 
         ^copied-math <number> 
         ^bar <bar> 
         ^child <child>)}

   (state <s3> ^result <r>)

   (interrupt)
