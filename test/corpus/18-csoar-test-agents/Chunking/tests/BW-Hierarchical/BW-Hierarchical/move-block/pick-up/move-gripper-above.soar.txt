   (state <s> ^name pick-up
              ^desired.moving-block <mblock>
              ^top-state <ts>)
   (<ts> ^clear <mblock>
         ^gripper <g>)
   (<g> ^above <> <mblock>
        ^position up
        ^holding nothing)
-->
   (<s> ^operator <op1> +)
   (<op1> ^name move-gripper-above
          ^destination <mblock>)
}

        ^position up
        ^holding <mblock>)
