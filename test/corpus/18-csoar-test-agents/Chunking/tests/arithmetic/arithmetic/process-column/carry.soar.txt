   (state <s> ^name << process-column get-digit1 >>
              ^carry 1
              ^current-column.next-column <nc>)

   (<op> ^name carry)

   (state <s> ^name << process-column get-digit1 >>

   (<op> ^name carry)
}
sp {process-column*apply*carry
   (state <s> ^operator.name carry
              ^carry 1
              ^current-column.next-column <nc>)
   (<nc> ^column t
         ^carry unknown)
-->
   (<nc> ^carry 1
         ^carry unknown -)
}

                        <op2> +)
   (<op1> ^name carry)
   (<op2> ^name write-result)
-->
   (<s> ^operator <op1> > <op2>)
}
