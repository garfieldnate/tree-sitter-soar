   :default
   (state <s> ^name selection
              ^evaluation <e>)
   (<e> ^estimated-cost <pc1> { <pc2> < <pc1> })
  -(<e> ^estimated-cost > <pc1>)
-->
   (<s> ^operator <op> + =, >)
   (<op> ^name remove-stale-estimated-cost
         ^evaluation <e>)
}

 #interrupt

   (<op> ^name remove-stale-estimated-cost
         ^evaluation <e>)
-->
#  (interrupt)
   (write (crlf) |Remove Stale Estimated Cost: | <e> |: | <pc2> | better: | <pc1>)
   (<e> ^estimated-cost <pc2> -)
}
