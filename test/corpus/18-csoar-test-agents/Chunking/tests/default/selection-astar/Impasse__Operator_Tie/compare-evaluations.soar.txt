   (state <s> ^name selection
              ^evaluation <e1> { <e2> <> <e1> })

         ^final-cost <fc>)

    (<com> ^op1 <so1>
           ^op2 <so2>)}

         ^op1 <so1>
         ^op2 <so2>)
