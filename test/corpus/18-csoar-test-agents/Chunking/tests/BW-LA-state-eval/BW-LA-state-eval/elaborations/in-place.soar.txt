   (state <s> ^desired.ontop <ontop-d>
              ^inplace-object <blockB>

              ^bottom-block <blockB>)
   (<ontop-s> ^top-block <blockA>
              ^bottom-block <blockB>)
-->
   (<s> ^inplace-object <blockA>)}

   (<ds> ^ontop <ontop1>)
   (<ontop1> ^top-block <blockA>
             ^bottom-block <blockB>)
